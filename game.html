using UnityEngine; using System.Collections.Generic; using System.Collections; public class GameController : MonoBehaviour { // --- 遊戲狀態機 --- public enum GameState { Login, Purchase, Lobby, Matching, Ready, Battle } public GameState currentState = GameState.Login; // --- 玩家與商店資料 --- [Header("Player Data")] public string playerName; public string email; public bool isAuthorized = false; [Header("Inventory")] public List guns = new List(100); // 100種槍 public List characters = new List(20); // 20種角色 [Header("Combat Stats")] public int ammo = 30; public float health = 100f; // --- 1. 系統流程 --- public void RegisterAndPurchase(string name, string mail, int planIndex, int payIndex) { // 模擬註冊與購買驗證 Debug.Log($"處理註冊: {name}, 方案: {planIndex}, 付款: {payIndex}"); string activationCode = "ACT-" + System.Guid.NewGuid().ToString().Substring(0, 8); Debug.Log($"自動生成啟動碼: {activationCode}"); isAuthorized = true; currentState = GameState.Lobby; } // --- 2. 大廳與賽季 --- public void EnterActivity(int activityIndex) { // 處理 10 種賽季活動規則 Debug.Log($"進入活動 {activityIndex}: 根據不同規則顯示福利"); } // --- 3. 配對與準備 --- public void StartMatchmaking() { StartCoroutine(MatchmakingRoutine()); } IEnumerator MatchmakingRoutine() { currentState = GameState.Matching; Debug.Log("上方顯示：配對中 (預計 10 秒)..."); yield return new WaitForSeconds(10f); currentState = GameState.Ready; Debug.Log("配對成功！請選擇角色與槍枝。"); } public void ConfirmReady() { currentState = GameState.Battle; Debug.Log("準備完成，遊戲開始！(100vs100)"); } // --- 4. 戰鬥邏輯 --- void Update() { if (currentState != GameState.Battle) return; // 操作：左鍵攻擊、右鍵裝彈、Q回血 if (Input.GetMouseButtonDown(0)) Attack(); if (Input.GetMouseButtonDown(1)) Reload(); if (Input.GetKeyDown(KeyCode.Q)) Heal(); // AI 協防 (簡單邏輯) CheckTeamSupport(); } void Attack() { ammo--; Debug.Log("攻擊，剩餘子彈: " + ammo); } void Reload() { ammo = 30; Debug.Log("裝填完畢"); } void Heal() { health += 20; Debug.Log("回血至: " + health); } void CheckTeamSupport() { // 當玩家危險時，隊友協助邏輯 if (health < 30) Debug.Log("偵測到危險，隊友前來協助！"); } }
