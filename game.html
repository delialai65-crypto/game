using UnityEngine; using System.Collections.Generic; public enum CPBLTeam { Elephants, // 台中大象 (兄弟) Monkeys, // 樂天桃子 (樂天) Dragons, // 台北飛龍 (味全) Lions, // 南方獅王 (統一) Guardians, // 新北悍將 (富邦) Hawks // 高雄雄鷹 (台鋼) } public enum Position { Pitcher, Catcher, Infield, Outfield } [CreateAssetMenu(fileName = "NewPlayer", menuName = "CPBL Game/Player Data")] public class PlayerStaticData : ScriptableObject { [Header("基本資料")] public string playerID; public string playerName; public CPBLTeam team; public Position position; [Header("打者打擊數值 (0-100)")] public float contact = 50f; // 影響打擊框大小 public float power = 50f; // 影響擊球初速 public float speed = 50f; // 影響跑壘與守備移速 [Header("投手投球數值 (0-100)")] public float velocity = 50f; // 基礎球速 public float control = 50f; // 控球精準度 public float stamina = 100f; // 體力值 /// /// 計算中職球隊天賦加成後的最終數值 /// public float GetModifiedStat(string statName, bool isHomeGame, int currentInning, int runnersOnBase) { float baseStat = 50f; switch (statName.ToLower()) { case "power": baseStat = power; break; case "contact": baseStat = contact; break; case "control": baseStat = control; break; case "speed": baseStat = speed; break; } // 六隊被動天賦邏輯實作 switch (team) { case CPBLTeam.Elephants: // 象魂：7局以後後半段比賽，打擊框(Contact)加成 10% if (currentInning >= 7 && statName.ToLower() == "contact") return baseStat * 1.10f; break; case CPBLTeam.Monkeys: // 暴力猿：只要壘有人，力量(Power)加成 8% if (runnersOnBase > 0 && statName.ToLower() == "power") return baseStat * 1.08f; break; case CPBLTeam.Lions: // 南霸天：若是主場作戰，投手控球(Control)提升 7% if (isHomeGame && statName.ToLower() == "control") return baseStat * 1.07f; break; case CPBLTeam.Guardians: // 悍將：雖然防守有時有梗，但危機處理強，滿壘時控球提升 5% if (runnersOnBase == 3 && statName.ToLower() == "control") return baseStat * 1.05f; break; } return baseStat; } }
